fn print;
  + PrintLen PrintLen PrintIter;
  while PrintIter not PrintLen do;
    outc VARS[PrintIter];
    incr PrintIter;
  end;
endf;
data match;
data no match;
set X 5;
in Y;
if X == Y do;
  set PrintIter 0;
  set PrintLen 5;
  print;
  outc 10;
  if 1 == 1 do;
    outc 65;
  endif;
endif;
if X != Y do;
  set PrintIter 5;
  set PrintLen 8;
  print;
  outc 10;
  if Y == 1 do;
    outc 66;
  endif;
endif;
